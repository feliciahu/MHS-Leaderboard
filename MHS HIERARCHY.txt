SELECT      
           SFDC.CUST_ACCT_ID,
           SFDC.SFDC_REP_SLS_TERR_ID,
           CUST.SLS_TERR_ID,
           CUST.HOME_DC_ID,
           SFDC.SLSPRSN_EMPLY_ID,
           EID.EMPLY_NAME, -- REQUIRED FIELD FOR FFA
           SFDC.SFDC_SLSPRSN_TITLE, -- REQUIRED FIELD FOR FFA
           SFDC.SPRVSR_EMPLY_ID AS SPRVSR_EMPLY_ID,
           EID_SPRVSR.EMPLY_NAME AS SPRVSR_EMPLY_NAME, -- REQUIRED FIELD FOR FFA
           EID_SPRVSR.JOB_TITLE_DSCR AS SPRVSR_EMPLY_TITLE,
           SFDC.SPRVSR_LVL_2_EMPLY_ID AS SPRVSR_LVL_2_EMPLY_ID,
           EID_SPRVSR_TWO.EMPLY_NAME AS SPRVSR_LVL_2_EMPLY_NAME, -- REQUIRED FIELD FOR FFA
           EID_SPRVSR_TWO.JOB_TITLE_DSCR AS SPRVSR_LVL_2_EMPLY_TITLE,
           SFDC.SPRVSR_LVL_3_EMPLY_ID AS SPRVSR_LVL_3_EMPLY_ID,
           EID_SPRVSR_THREE.EMPLY_NAME AS SPRVSR_LVL_3_EMPLY_NAME,
           EID_SPRVSR_THREE.JOB_TITLE_DSCR AS SPRVSR_LVL_3_EMPLY_TITLE
FROM        PRD_PSAS_DB.APP.V_IF_RA_FACT_SFDC_REP_CUST_ACCT_CURR SFDC -- main table
JOIN        PRD_PSAS_DB.APP.V_IF_RA_DIM_CUST_ACCT_CURR CUST
ON          SFDC.CUST_ACCT_ID = CUST.CUST_ACCT_ID
LEFT JOIN   PRD_PSAS_DB.APP.V_IF_PHARMA_DIM_EMPLY_EID EID
ON          SFDC.SLSPRSN_EMPLY_ID = EID.EMPLY_ID
LEFT JOIN   PRD_PSAS_DB.APP.V_IF_PHARMA_DIM_EMPLY_EID EID_SPRVSR
ON          SFDC.SPRVSR_EMPLY_ID = EID_SPRVSR.EMPLY_ID
LEFT JOIN   PRD_PSAS_DB.APP.V_IF_PHARMA_DIM_EMPLY_EID EID_SPRVSR_TWO
ON          SFDC.SPRVSR_LVL_2_EMPLY_ID = EID_SPRVSR_TWO.EMPLY_ID
LEFT JOIN   PRD_PSAS_DB.APP.V_IF_PHARMA_DIM_EMPLY_EID EID_SPRVSR_THREE
ON          SFDC.SPRVSR_LVL_3_EMPLY_ID = EID_SPRVSR_THREE.EMPLY_ID
WHERE       SFDC.SFDC_REP_SLS_TERR_ID not in ('N/A') 
                                            --AND SFDC.SFDC_SLSPRSN_TITLE in ('AM','RSM','FSL','KAE','SSM') 
                                            --AND  CUST.ACTIVE_CUST_IND = 'A'
 
 
order by  SFDC.CUST_ACCT_ID, SFDC.SFDC_REP_SLS_TERR_ID, SFDC.SLSPRSN_EMPLY_ID, SFDC.SPRVSR_EMPLY_ID;
